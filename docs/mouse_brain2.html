jaccard_index <- function(vec1, vec2) {
  
  set1 <- unique(vec1)
  
  set2 <- unique(vec2)
  
  intersection <- length(intersect(set1, set2))
  
  union <- length(union(set1, set2))
  
  return(intersection / union)
  
}

hoyer_sparsity <- function(x) {
  
  n <- length(x)
  (1 / (sqrt(n) - 1)) * (sqrt(n) - (sum(x) / (sqrt(sum(x ^ 2)))))
  
}
